/*C++ program of Number Guessing Game */

#include <iostream>
#include <cstdlib>
using namespace std;

//Generating a random number 
int Random_Num_generator(){
    srand(time(0));
    return (rand()%100);
}

//function of guessing number which is use to guess the number 
void Guessing_Number(int random_num){
    int num;
    //taking input from the user
    cout << "Guess a number  : ";
    cin >> num;
    
    //using if-else statement to solve guessing number problem
    if (random_num == num){
        cout << "Number is Matched! Random number generated by computer is : "<<random_num<<endl;
        cout << "Thank You!"<<endl;
    }
    else if (random_num > num){
        cout << "Your Random number is large! insert some larger number.\n"<<endl;
        Guessing_Number(random_num);//Recursion uses
    }
    else{
        cout << "Your Random number is small! insert some smaller number.\n"<<endl;
        Guessing_Number(random_num);//Recursion uses
    }
    
}

//Driver code
int main()
{
    cout << "Guess number under 100! ";
    //function call
    int random_num = Random_Num_generator();
    Guessing_Number(random_num);

    return 0;
}
